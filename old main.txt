int readOperand(char *operand, char n, char *op)
{
	char inp='0xFF';
	while(1)
	{
		do
		{
			inp=keypad_u8read();
		} while(inp==0xFF);
		_delay_ms(250);
		if(inp<='9' && inp>='0')
		{
			(*operand) = inp;
			LCD_vSend_char(*operand);
		}
		else
		{
			(*op) = inp;
			break;
		}
		operand++;
		n++;
	}
	operand = operand-n;
	int value = charArrayToInt(operand,n);
	return value;
}

int operandtoInteger(char *operand, char n)
{
	int returnval=0;
	char i=0;
	for(i=0;i<n;i++)
	{
		returnval = returnval*10 + ((*(operand+i))-'0');
	}
	return returnval;
}


void printInt(int z)
{
	LCD_vSend_char((z/100)+48);
	LCD_vSend_char(((z/10)%10)+48);
	LCD_vSend_char((z%10) + 48);
}

int calcResult(int x, int y, char op)
{
	if(op=='+')
	{
		return x+y;
	}
	else if(op=='-')
	{
		return x-y;
	}
	else if(op=='x')
	{
		return x*y;
	}
	else if(op='/')
	{
		return x/y;
	}
	else
	{
		LCD_clearscreen();
	}
}
